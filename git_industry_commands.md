
## 1. git config --global user.name
### Syntax:
git config --global user.name "Your Name"
### Purpose:
Sets global username for Git.
### Example:
git config --global user.name "meghana-221116"
## 2. git config --global user.email
### Syntax:
git config --global user.email "your@email.com"
### Purpose:
Sets global email for Git.
### Example:
git config --global user.email "meghana@gmail.com"
## 3. git status
### Syntax:
git status
### Purpose:
Shows current status of repository.
### Example:
git status
## 4.git config --list
### Syntax:
git config --list
### Purpose:
Shows all git configuration settings.
### Example:
git config --list
## 5.git config --unset user.name
### Syntax:
git config --unset user.name
### Purpose:
Removes a configuration value.
### Example:
git config --unset user.name
